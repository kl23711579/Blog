7. Structrue ¶

package main

import "fmt"

type Vertex struct {
    X int
    Y int
}

func main() {
    fmt.Println(Vertex{1, 2})

    v := Vertex{1, 2}
    v.X = 4
    fmt.Println(v.X)
}

// result
{1 2}
4

package main

import "fmt"

type Vertex struct {
    X, Y int
}

var (
    v1 = Vertex{1, 2}
    v2 = Vertex{X: 1} // Y: 0 is implicit
    v3 = Vertex{}     // X: 0 Y: 0
    p = &Vertex{1, 2}
)

func main(){
    fmt.Println(v1, v2, v3, p)
}

// Result
{1 2} {1 0} {0 0} &{1 2}

s := []struct{
    i int
    b bool
}{
    {1, true},
    {2, false},
    {3, true},
    {4, true},
}

8. Array ¶

The type [n]T is an array of n values of type T

var a [2]string
a[0] = "hello"
a[1] = "world!"
fmt.Println(a[0], a[1])
fmt.Println(a)

primes := [6]int{2, 3, 5, 7, 11, 13}
fmt.Println(primes)

// Result
Hello World
[Hello World]
[2 3 5 7 11 13]
